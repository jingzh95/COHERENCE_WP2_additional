# 9 files each
 # risk score
score <- list.files(pattern = "Output_disease_risk_score_weights_AllWaves")
 # find the min number of predictors 
num_pred <- vector("integer", length(score))
for ( j in seq_along(num_pred)){
  num_pred[j] <- nrow(read.xlsx(score[j]))
  if (j == length(score)) num_pred <- min(num_pred)
}

 # model with coefficients
class0 <- list.files(pattern = "class0_summaryAllWaves")

for ( i in seq_along(score)){
  df <- read.xlsx(score[i])
  df$Ensamble <- as.numeric(df$Ensamble)
  df$Weights <- as.integer(df$Weights)
  # keep the first 7
  df <- df[1:num_pred,]
  
  # probability
  eqn <- read_csv(class0[i])
  if (i==1|i==2){
    b0 <- as.numeric(str_split(eqn[7,], " ")[[1]][3])
    b1 <- as.numeric(str_split(eqn[8,], " ")[[1]][8])
  }else{
    b0 <- as.numeric(str_split(eqn[7,], " ")[[1]][2]) # intercept
    b1 <- as.numeric(str_split(eqn[8,], " ")[[1]][9]) # slop
  }
  
  prob <- function(x){
   1/(1+exp(-b0-b1*x))
  }
  df$eqn <- paste0("1/(1+exp(-(",as.character(b0),"+", as.character(b1), "*DRS)))")
  df$Probability <- prob(as.numeric(df$Ensamble))
  
  # round the score and probability? 
  # df$Ensamble <- round(as.numeric(df$Ensamble), 2) 
  setnames(df, "Ensamble", "DRS")
  df <- df[!df$Predictors %like% "var4",]
   if (i == 9){
     df$age_group <- "all"
     df$gender <- "both"
   }else{
    df$age_group <- str_split(score[i],"_")[[1]][7]
    df$gender <- substr(str_split(score[i], "_")[[1]][8], 1, 1)
   }
  if (i == 1) dt <- df
  else dt <- rbind(dt, df)
}
